65 inline std::string thousandSeparator(T n, int decimals = 3) {
66 // Get the sign of the number and remove it
67 int sign = (n < 0) ? -1 : 1;
68 double approx = power((double)10.0, decimals);
69 n *= sign;
70 // Get the integer part of the number
71 long long intPart = (long long)n;
72 // Check if the fractional part has any value
73 bool hasFracPart = (n - intPart > 0);
74 // Get the fractional part of the number and trim it to n decimal places
75 double fracPart = round((n - intPart) * (approx)) / (approx);
76 // Create a locale object to represent a specific localization
77 locale loc("");
78 // Set the thousand separator according to the localization
79 char separator = use_facet< numpunct<char> >(loc).thousands_sep();
80 // Convert the integer part to a string
81 string intStr = to_string(intPart);
82 // Insert the separator every 3 digits from the end
83 for (int i = intStr.length() - 3; i > 0; i -= 3) {
84 intStr.insert(i, 1, separator);
85 }
86 // Convert the fractional part to a string and trim it to n decimal places
87 stringstream stream;
88 if (hasFracPart) {
89 stream << fixed << setprecision(decimals) << fracPart;
90 }
91 string fracStr = (hasFracPart) ? stream.str().substr(1) : "";
92 // Combine the integer and fractional parts into a single string
93 string result = intStr + fracStr;
94 // Add the sign to the beginning of the string if the number was negative
95 if (sign == -1) {
96 result = "-" + result;
97 }
98 return result;
99 }
