38 inline double power(double base, int exponent) {
39 if (exponent == 0) {
40 return 1.0;

42 else if (exponent < 0) {
43 return 1.0 / power(base, -exponent);

68 inline std::string thousandSeparator(T n, int decimals = 3) {
69 // Get the sign of the number and remove it
70 int sign = (n < 0) ? -1 : 1;
71 double approx = power((double)10.0, decimals);
72 n *= sign;
73 // Get the integer part of the number
74 long long intPart = (long long)n;
75 // Check if the fractional part has any value
76 bool hasFracPart = (n - intPart > 0);
77 // Get the fractional part of the number and trim it to n decimal places
78 double fracPart = round((n - intPart) * (approx)) / (approx);
79 // Create a locale object to represent a specific localization
80 locale loc("");
81 // Set the thousand separator according to the localization
82 char separator = use_facet< numpunct<char> >(loc).thousands_sep();
83 // Convert the integer part to a string
84 string intStr = to_string(intPart);
85 // Insert the separator every 3 digits from the end
86 for (int i = intStr.length() - 3; i > 0; i -= 3) {
87 intStr.insert(i, 1, separator);
88 }
89 // Convert the fractional part to a string and trim it to n decimal places
90 stringstream stream;
91 if (hasFracPart) {
92 stream << fixed << setprecision(decimals) << fracPart;
93 }
94 string fracStr = (hasFracPart) ? stream.str().substr(1) : "";
95 // Combine the integer and fractional parts into a single string
96 string result = intStr + fracStr;
97 // Add the sign to the beginning of the string if the number was negative
98 if (sign == -1) {
99 result = "-" + result;
100 }
101 return result;
102 }
