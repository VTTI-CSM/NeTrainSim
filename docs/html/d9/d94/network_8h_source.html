270 bool isConflictZone(std::shared_ptr<Train> train,
271 std::shared_ptr<NetNode> node1,
272 std::shared_ptr<NetNode> node2) {
273 // Get indices of nodes in train path
274 int n1 = train->trainPathNodes.index(node1);
275 int n2 = train->trainPathNodes.index(node2);
276
277 // Ensure n1 is less than n2
278 if (n1 > n2) { swap(n1, n2); }
279
280 for (int i = n1; i < n2; i++) {
281 // Get links between consecutive nodes
282 int nextI = i + 1;
283 Vector<std::shared_ptr<NetLink>> l1 =
284 train->trainPathNodes.at(
285 i)->linkTo[train->trainPathNodes.at(nextI)];
286 Vector<std::shared_ptr<NetLink>> l2 =
287 train->trainPathNodes.at(
288 nextI)->linkTo[train->trainPathNodes.at(i)];
289
290 // Add links to a set
291 std::set<std::shared_ptr<NetLink>> temp;
292 if (l1.size() > 0) { temp.insert(l1.begin(), l1.end()); }
293 if (l2.size() > 0){ temp.insert(l2.begin(), l2.end()); }
294
295 // Check if only one link exists
296 if (temp.size() <= 1) { return true; }
297 }
298
299 // If more than one link exists between any pair of nodes, return false
300 return false;
301 }

316 bool DistanceToEndOfAllLinkTrainsIsLarge(
317 const std::shared_ptr <NetLink> link,
318 const std::shared_ptr <Train> train)
319 {
320 // Store distances to other trains on the same link
321 std::set<double> distanceToOtherTrains;
322
323 // Compute distances for each train on the link
324 for (std::shared_ptr <Train>& t : link->currentTrains) {
325 if (t != train) {
326 distanceToOtherTrains.insert(
327 Utils::getDistanceByTwoCoordinates(
328 t->startEndPoints[1],
329 train->currentCoordinates));
330 }
331 }
332 // Check if any distance is less than 2
333 return std::find_if(distanceToOtherTrains.begin(),
334 distanceToOtherTrains.end(),
335 [](int v) {
336 return v < 2;
337 }) == distanceToOtherTrains.end();
338 }

372 pair<double, double> getPositionbyTravelledDistance(
373 std::shared_ptr <Train> train,
374 double &travelledDistance) {
375
376 // If the train has not moved, its position is the coordinates of the
377 // first node in its path.
378 if (travelledDistance <= 0.0) {
379 return train->trainPathNodes[0]->coordinates();
380 }
381
382 // If the train has moved more than or equal to its total path length,
383 // its position is the coordinates of the last node in its path.
384 if (travelledDistance >= train->trainTotalPathLength) {
385 return train->trainPathNodes.back()->coordinates();
386 }
387
388 // Traverse the train path
389 for (int i = 0; i < train->trainPath.size() - 1; i++) {
390 // Check if travelled distance is between the cumulative
391 // lengths of consecutive links
392 if (travelledDistance > train->linksCumLengths[i] &&
393 (i == train->trainPath.size() - 2 ||
394 travelledDistance <= train->linksCumLengths[i+1])) {
395 int nextI = i + 1;
396
397 // Get the start node and the link on which the train is
398 // currently on
399 std::shared_ptr<NetNode> startNode = train->trainPathNodes[i];
400 std::shared_ptr<NetLink> link =
401 getLinkByStartandEndNodeID(train, train->trainPath[i],
402 train->trainPath[nextI], true);
403
404 // Calculate the distance travelled on the current link
405 double travelledDistanceOnLink =
406 travelledDistance - train->linksCumLengths[i];
407
408 // Calculate and return the position on the current vector
409 // based on the distance travelled on the link
410 return link->findPositionOnLink(travelledDistanceOnLink,
411 startNode);
412 }
413 }
414 // If travelledDistance does not fall within any link's cumulative
415 // lengths, return (0.0, 0.0)
416 return std::make_pair(0.0, 0.0);
417 }

491 std::shared_ptr<NetLink> getLinkByStartandEndNodeID(
492 const std::shared_ptr<Train> train,
493 int startID,
494 int endID, bool calcExact = true) {
495 Vector<std::shared_ptr<NetLink>> betweenNodesLinks =
496 this->getNodeByID(startID)->linkTo.at(this->getNodeByID(endID));
497 if (betweenNodesLinks.size() == 1) {
498 return betweenNodesLinks.at(0);
499 }
500
501 // If there is more than one link, find the one that currently contains
502 // the train
503 for (std::shared_ptr<NetLink> l : betweenNodesLinks) {
504 if (l->currentTrains.exist(train)) {
505 return l;
506 }
507 }
508
509 // If the train is not on any of the links, sort the links by cost and
510 // potentially return the one with the lowest cost
511 Vector<std::shared_ptr<NetLink>> betweenNLinks(
512 betweenNodesLinks.begin(),
513 betweenNodesLinks.end());
514 std::sort(betweenNLinks.begin(), betweenNLinks.end(),
515 [](std::shared_ptr<NetLink> a, std::shared_ptr<NetLink> b) {
516 return a->cost < b->cost; });
517 if (!calcExact) {
518 return betweenNLinks[0];
519 }
520
521 // If calcExact is true, find a link where the train can travel without
522 // colliding with other trains
523 for (std::shared_ptr <NetLink> l : betweenNLinks) {
524 if (l->currentTrains.empty() || (isSameDirection(l, train) &&
525 DistanceToEndOfAllLinkTrainsIsLarge(l, train))) {
526 return l;
527 }
528 }
529 return betweenNLinks[0];

824 std::pair<Vector<std::shared_ptr<NetNode>>, double> shortestPathSearch(
825 int startNodeID, int targetNodeID) {
826 // clear all the nodes for the new sheart path
827 for (auto& n : this->nodes) {
828 n.second->clearGraphSearchParams();
829 }
830 // get the nodes by their user id's
831 std::shared_ptr<NetNode> startNode = this->getNodeByID(startNodeID);
832 std::shared_ptr<NetNode> targetNode = this->getNodeByID(targetNodeID);
833 // override the inf value of the start node
834 startNode->graphSearchDistanceFromStart = 0.0;
835
836 // keep looping until the target node is reached
837 while (! targetNode->graphSearchVisited) {
838 // get the min-distance-unvisited node
839 std::shared_ptr<NetNode> currentNode = minimumDistance();
840 // if nothing was found, break
841 if (currentNode == nullptr) { break; }
842 // set the node as visited
843 currentNode->graphSearchVisited = true;
844 // check all the connected links/nodes
845 for (auto& connectedNode : currentNode->getNeighbors()) {
846 // check if a new node has a lower value
847 if ((currentNode->graphSearchDistanceFromStart +
848 currentNode->linkTo.at(connectedNode).at(0)->length) <
849 connectedNode->graphSearchDistanceFromStart) {
850 // set the value
851 connectedNode->graphSearchDistanceFromStart =
852 currentNode->graphSearchDistanceFromStart +
853 currentNode->linkTo.at(connectedNode).at(0)->length;
854 connectedNode->graphSearchPreviousNode = currentNode;
855 }
856 }
857 }
858
859 Vector<std::shared_ptr<NetNode>> path;
860 path.push_back(targetNode);
861 while (true) {
862 std::shared_ptr<NetNode> node = path.back();
863 if (node->graphSearchPreviousNode == nullptr) { break; }
864 //if (path.exist(node->graphSearchPreviousNode)) { break; }
865 path.push_back(node->graphSearchPreviousNode);
866 }
867 std::reverse(path.begin(), path.end());
868 return std::make_pair(path, targetNode->graphSearchDistanceFromStart);
869 }
